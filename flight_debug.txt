-- ===== FLIGHT =====
local bodyVel = nil
local cloudTimer = 0
local coinTimer = 0

local function stopFlying()
	isFlying = false
	_G.isFlying = false
	local char = player.Character
	if char then
		local hrp = char:FindFirstChild("HumanoidRootPart")
		local hum = char:FindFirstChild("Humanoid")
		if bodyVel then bodyVel:Destroy() end
		if hrp then
			local old = hrp:FindFirstChild("FartVelocity")
			if old then old:Destroy() end
		end
		if hum then hum.PlatformStand = false end
	end
	bodyVel = nil
	flightStatsFrame.Visible = false; windIndicatorFrame.Visible = false
	if _G.checkMilestones then _G.checkMilestones() end
	_G.peakHeight = 0; _G.ringsCollectedFlight = 0
	updateFartBtn()
end
_G.stopFlying = stopFlying

local function startFlying()
	if isFlying then return end
	if gasMeter <= 0 then return end
	if not hasBoughtFood then return end
	local char = player.Character
	if not char then return end
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not hrp then return end
	local hum = char:FindFirstChild("Humanoid")

	isFlying = true
	_G.isFlying = true
	flightStartY = hrp.Position.Y; flightStartTime = tick()
	_G.peakHeight = hrp.Position.Y; _G.ringsCollectedFlight = 0

	if bodyVel then bodyVel:Destroy() end
	local old = hrp:FindFirstChild("FartVelocity")
	if old then old:Destroy() end

	if hum then hum.PlatformStand = true end

	bodyVel = Instance.new("BodyVelocity")
	bodyVel.Name = "FartVelocity"
	bodyVel.MaxForce = Vector3.new(0, 1e6, 0)
	bodyVel.Velocity = Vector3.new(0, 60, 0)
	bodyVel.Parent = hrp

	updateFartBtn()
end

player.CharacterAdded:Connect(function(char)
	isFlying = false; _G.isFlying = false
	if bodyVel then bodyVel:Destroy(); bodyVel = nil end
	gasMeter = 0; hasBoughtFood = false; _G.hasLanded = true
	updateMeter(); updateFartBtn()
	onLand(char)
end)
onLand(character)

-- ===== HEARTBEAT =====
RunService.Heartbeat:Connect(function(dt)
	if twoXBoostActive and os.time() > twoXBoostEndTime then twoXBoostActive = false end
	if not isFlying then return end
	local char = player.Character
	if not char then stopFlying(); return end
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not hrp then stopFlying(); return end

	local power = 0
	pcall(function() if leaderstats then local t = leaderstats:FindFirstChild("TotalFartPower"); if t then power = t.Value end end end)

	local speed = 30 + (power * 0.05)
	if speed > 200 then speed = 200 end
	local cap = 100 + (power * 10)
	speed = speed * _G.serverEventSpeedMult
	cap = cap * _G.serverEventHeightMult
	if twoXBoostActive then speed = speed * 2 end

	if not bodyVel or not bodyVel.Parent then
		bodyVel = Instance.new("BodyVelocity")
		bodyVel.Name = "FartVelocity"
		bodyVel.MaxForce = Vector3.new(0, 1e6, 0)
		bodyVel.Parent = hrp
	end

	if hrp.Position.Y < cap then
		bodyVel.Velocity = Vector3.new(0, speed, 0)
	else
		bodyVel.Velocity = Vector3.new(0, 0, 0)
	end

	gasMeter = math.max(0, gasMeter - 4 * dt)
	updateMeter()

	if hrp.Position.Y > _G.peakHeight then _G.peakHeight = hrp.Position.Y end

	flightStatsFrame.Visible = true
	fsHeight.Text = "\xF0\x9F\x93\x8F Height: " .. math.floor(hrp.Position.Y)
	fsRings.Text = "\xF0\x9F\x92\x8D Rings: " .. _G.ringsCollectedFlight .. " (x" .. string.format("%.1f", ringMultiplier) .. ")"
	fsAir.Text = "\xe2\x8f\xb1 Air: " .. math.floor(tick() - flightStartTime) .. "s"

	coinTimer = coinTimer + dt
	if coinTimer >= 0.2 then
		coinTimer = 0
		local height = math.max(0, hrp.Position.Y - 5)
		local coins = math.floor(height / 10) * 0.15 * _G.serverEventCoinMult
		if coins > 0 then pcall(function() CoinEvent:FireServer(coins) end) end
	end

	if gasMeter <= 0 then stopFlying(); return end

	-- ring collection
	for i = #_G.activeRings, 1, -1 do
		local r = _G.activeRings[i]
		if r.part and r.part.Parent then
			if (hrp.Position - r.part.Position).Magnitude < 16 then
				local rpos, rcol, ridx, rdir = r.pos, r.color, r.idx, r.dir
				r.part:Destroy(); table.remove(_G.activeRings, i)
				ringStreak = ringStreak + 1; ringMultiplier = 1 + ringStreak * 0.2
				local bonus = math.floor(15 * ringMultiplier * _G.serverEventRingMult)
				_G.ringsCollectedFlight = _G.ringsCollectedFlight + 1
				if CoinEvent then pcall(function() CoinEvent:FireServer(bonus) end) end
				showFloatingText("+" .. bonus .. " \xF0\x9F\xAA\x99 x" .. string.format("%.1f", ringMultiplier), Color3.fromRGB(255, 215, 0))
